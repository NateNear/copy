```jsx copy
clear all
clc
j=sqrt(-1);
vt=34.64*(10^3);
p=48*(10^6);
pf=0.8;
xs=10;
vtph=vt/(sqrt(3));
phi=acos(pf);
q=p*tan(phi);
i=(p-j*q)/(3*vtph);
e=vtph+ (j*i*xs);
ef=abs(e)
delta=0:1:180;
del=(delta*3.142)/180;
p1=(vtph*ef*sin(del))/xs;
reg=((ef-vtph)/vtph)*100
plot(delta,p1)
```
